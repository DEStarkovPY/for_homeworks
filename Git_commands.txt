git branch <name> - введение новой ветки
git branch - просмотр веток
git checkout <name> - переключение на данную ветку
git push -u <name del repositor> <name local repositor> - добавление ветки на удалённый репозиторий
git rm --cached <name> -удаление файла
git merge --no-ff <branch name> -m <comment> - сохранение информации с ветки 
git log --oneline --graph - история коммитов
git add/ git merge --abort - решение конфликта при слиянии 
git pull <name del repositor> - получение обновлений в файле с удалённого репозитория
